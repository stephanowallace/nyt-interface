class ArticleImageUtil {
  static formatImage(imageObject) {
    return {
      alt: imageObject.caption,
      src: imageObject.url,
    };
  }
 
  static getImageBySize(arrayOfImages, size) {
    const foundImageArray = arrayOfImages.filter(image => image.format === size);
    return foundImageArray.length
      ? this.formatImage(foundImageArray[0])
      : null;
  }
}
 
export default ArticleImageUtil;
