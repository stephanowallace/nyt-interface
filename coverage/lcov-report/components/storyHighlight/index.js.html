import React, { Component } from 'react';
import PropTypes from 'prop-types';
 
class StoryHighlight extends Component {
  renderAuthorAndCreationDate() {
    const { author, publishedData } = this.props;
    return (
      <p>
        <span>{publishedData}</span>
        <span>{author}</span>
      </p>
    );
  }
 
  renderTitle() {
    const { onClick, title } = this.props;
    return (
      <h2 onClick={onClick}>{title}</h2>
    );
  }
 
  render() {
    return (
      <div>
        {this.renderTitle()}
        {this.renderAuthorAndCreationDate()}
      </div>
    );
  }
}
 
StoryHighlight.propTypes = {
  author: PropTypes.string.isRequired,
  onClick: PropTypes.func.isRequired,
  publishedData: PropTypes.string.isRequired,
  title: PropTypes.string.isRequired,
};
 
export default StoryHighlight;
