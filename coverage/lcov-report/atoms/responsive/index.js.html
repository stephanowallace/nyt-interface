import React, { Component } from 'react';
import PropTypes from 'prop-types';
import Constants from 'Constants';
import ResponsiveMode from './responsiveMode';
 
const { Provider, Consumer } = React.createContext();
 
class Responsive extends Component {
  constructor() {
    super();
    this.timeout = null;
    this.state = { mode: null };
    this.Mode = ResponsiveMode;
    this.Consumer = Consumer;
    this.PHONE = 'phone';
    this.TABLET = 'tablet';
    this.DESKTOP = 'desktop';
  }
 
  componentDidMount() {
    this.setMode();
    window.addEventListener('resize', this.onResize);
  }
 
  componentWillUnmount() {
    window.removeEventListener('resize', this.onResize);
  }
 
  onResize() {
    if (this.timeout) clearTimeout(this.timeout);
    this.timeout = setTimeout(this.setMode, 300);
  }
 
  getScreenMode() {
    const screenWidth = document.body.offsetWidth;
    const { sizes } = this.props;
    if (screenWidth < sizes.small) return Responsive.PHONE;
    if (screenWidth >= sizes.small && screenWidth < sizes.medium) {
      return Responsive.TABLET;
    }
    return Responsive.DESKTOP;
  }
 
  setMode() {
    const mode = this.getScreenMode();
    this.setState({ mode });
  }
 
  getContext() {
    return { ...this.state };
  }
 
  render() {
    const { children } = this.props;
    const context = this.getContext();
    return (
      <Provider value={context}>
        {children}
      </Provider>
    );
  }
}
 
Responsive.propTypes = {
  children: PropTypes.oneOfType([
    PropTypes.array,
    PropTypes.number,
    PropTypes.object,
    PropTypes.string,
  ]).isRequired,
  sizes: PropTypes.shape({
    extraSmall: PropTypes.number.isRequired,
    small: PropTypes.number.isRequired,
    medium: PropTypes.number.isRequired,
    large: PropTypes.number.isRequired,
  }),
};
 
Responsive.defaultProps = {
  sizes: {
    extraSmall: Constants.SCREEN_SIZES.EXTRA_SMALL,
    small: Constants.SCREEN_SIZES.SMALL,
    medium: Constants.SCREEN_SIZES.MEDIUM,
    large: Constants.SCREEN_SIZES.LARGE,
  },
};
 
 
export default Responsive;
