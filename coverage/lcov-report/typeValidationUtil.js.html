export const validateString = (valueToBeTested, callback) => {
  try {
    if (typeof valueToBeTested === 'string') return callback();
    throw new TypeError('Expected string');
  } catch (e) { throw e; }
};
 
export const validateObject = (valueToBeTested, callback) => {
  try {
    if (typeof valueToBeTested === 'object') return callback();
    throw new Error('Expected object');
  } catch (e) { throw e; }
};
